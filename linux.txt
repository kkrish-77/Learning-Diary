ls # Listing object 
ls -a # Listing all the object including hidden files
ls -lt # List
ls -ltr > 
uptime -h > 
free -h > 
less <Filename> ... then it will open the editor and then search using /<word>  # For looking for an specific word in the file 
less <Filename> ... then it will open the editor and then search using ?<word>  # For looking for an specific word in the file in reverse order
touch <filename> # For creating an file
rm <filename> # For deleting an file
mkdir <directory name> # For creating an directory
rmdir <directory name> # For deleting an directory
pwd # Print working directory 
cd # Change directory 
cd .. # Going back to previous directory
cd ../.. # Going back x2
cp <source> <destination> # copy the contents and can also be used for backup of one file to other
mv <filename> <destination> # For cut pasting the file 
mv <old-filename> <new-filename> # For renaming the file
head -n 5 <filename> # First 5 lines of contents in the file
tail -n 5 <filename> # Last 5 lines of contents in the file
sort <filename> # Sort
sort -r <filename> # Reverse sort 
sort <filename> | uniq # Filtering only unique values in the file and removing duplicate values
